async function salvarConfig() { const nomeEvento = document.getElementById('conf-nome').value.trim(); if(!nomeEvento) return alert("Digite o nome do evento."); // Converte MM:SS para segundos totais antes de salvar const m = parseInt(document.getElementById('trim-min').value) || 0; const s = parseInt(document.getElementById('trim-sec').value) || 0; const totalSec = (m * 60) + s; // --- INICIO DA CORREÇÃO (Mantendo o ID Fixo) --- const configLocalAnterior = JSON.parse(localStorage.getItem('config_atual')) || {}; // --- FIM DA CORREÇÃO --- const config = { id: document.getElementById('conf-id').value || String(Date.now()), nome: nomeEvento, // --- INICIO DA CORREÇÃO (Vinculando o ID Fixo) --- id_fixo: configLocalAnterior.id_fixo || nomeEvento, // --- FIM DA CORREÇÃO --- data: document.getElementById('conf-data').value, tempo: String(document.getElementById('conf-tempo').value), preset: document.getElementById('conf-preset').value, v1: document.getElementById('conf-v1').value, v2: document.getElementById('conf-v2').value, v3: document.getElementById('conf-v3').value, reverso: document.getElementById('conf-reverso').checked, molduraUrl: document.getElementById('conf-moldura-url').value, musicaUrl: document.getElementById('conf-musica-url').value, musicaName: document.getElementById('conf-musica-name').value, musicaTrim: String(totalSec), musicaLoopVol: document.getElementById('conf-musica-loop-vol').checked }; try { const res = await fetch(`${WORKER_URL}/salvar-evento`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(config) }); if(res.ok) { // --- INICIO DA CORREÇÃO (Avisa o navegador que o nome mudou) --- localStorage.setItem('config_atual', JSON.stringify(config)); // --- FIM DA CORREÇÃO --- alert("Evento salvo!"); window.location.href = 'index.html'; } } catch (e) { alert("Erro ao salvar."); } }
